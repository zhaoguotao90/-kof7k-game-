const testVideoBtn = document.getElementById('test-video-btn'); const rewardVideo = document.getElementById('reward-video'); const closeBtn = document.getElementById('close-btn'); const fullscreenBtn = document.getElementById('fullscreen-btn'); let rufflePlayer = null; let isMuted = false; // 更新进度条 function updateProgress(percent) { progressBar.style.width = percent + '%'; loadingText.textContent = `加载中... ${Math.round(percent)}%`; } // 加载游戏函数 async function loadGame() { updateProgress(0); loadingOverlay.style.display = 'flex'; errorMessage.style.display = 'none'; try { // 加载Ruffle播放器 if (!window.RufflePlayer) { loadingText.textContent = '正在加载Ruffle引擎...'; await new Promise((resolve) => { const script = document.createElement('script'); script.src = 'ruffle.js'; script.onload = resolve; document.head.appendChild(script); }); updateProgress(30); } // 创建播放器实例 const ruffle = window.RufflePlayer.newest(); rufflePlayer = ruffle.createPlayer(); gameContainer.appendChild(rufflePlayer); updateProgress(50); // 加载Base64分块文件 loadingText.textContent = '正在加载游戏分块...'; const totalParts = 8; // 根据实际分块数量调整 let base64Data = ''; for (let i = 1; i <= totalParts; i++) { const response = await fetch(`game_part_${i}.b64`); const text = await response.text(); base64Data += text; updateProgress(50 + (i / totalParts) * 40); } // 转换Base64为二进制 loadingText.textContent = '正在解码游戏数据...'; const binary = atob(base64Data); const uintArray = new Uint8Array(binary.length); for (let i = 0; i < binary.length; i++) { uintArray[i] = binary.charCodeAt(i); } updateProgress(95); // 加载游戏 await rufflePlayer.load({ data: uintArray.buffer, width: "100%", height: "100%", autoplay: "on" }); // 加载成功 updateProgress(100); setTimeout(() => { loadingOverlay.style.opacity = '0'; setTimeout(() => { loadingOverlay.style.display = 'none'; }, 500); }, 500); } catch (error) { console.error('加载失败:', error); handleError(error); } } // 错误处理 function handleError(error) { let errorMsg = ''; if (error.message.includes('fetch')) { errorMsg = '分块文件加载失败，请检查文件路径和网络连接'; } else if (error.message.includes('Ruffle')) { errorMsg = 'Ruffle播放器加载失败'; } else { errorMsg = error.message; } errorMessage.textContent = errorMsg; errorMessage.style.display = 'block'; reloadBtn.style.display = 'block'; } // 通关奖励 function showRewardVideo() { gameContainer.style.display = 'none'; rewardVideo.style.display = 'block'; closeBtn.style.display = 'block'; fullscreenBtn.style.display = 'block'; rewardVideo.play().catch(e => { console.log('视频自动播放失败:', e); rewardVideo.controls = true; }); } // 返回游戏 function backToGame() { rewardVideo.style.display = 'none'; closeBtn.style.display = 'none'; fullscreenBtn.style.display = 'none'; rewardVideo.pause(); gameContainer.style.display = 'flex'; } // 切换音量 function toggleVolume() { isMuted = !isMuted; if (rufflePlayer) { rufflePlayer.muted = isMuted; } volumeBtn.innerHTML = isMuted ? '🔇 音量' : '🔊 音量'; } // 全屏播放视频 function toggleFullscreen() { if (rewardVideo.requestFullscreen) { rewardVideo.requestFullscreen(); } else if (rewardVideo.mozRequestFullScreen) { rewardVideo.mozRequestFullScreen(); } else if (rewardVideo.webkitRequestFullscreen) { rewardVideo.webkitRequestFullscreen(); } else if (rewardVideo.msRequestFullscreen) { rewardVideo.msRequestFullscreen(); } } // 事件监听 reloadBtn.addEventListener('click', loadGame); volumeBtn.addEventListener('click', toggleVolume); testVideoBtn.addEventListener('click', showRewardVideo); closeBtn.addEventListener('click', backToGame); fullscreenBtn.addEventListener('click', toggleFullscreen); // 监听键盘事件 document.addEventListener('keydown', (e) => { if (e.key === 'Escape') { backToGame(); } }); // 页面加载后自动开始加载游戏 window.addEventListener('DOMContentLoaded', loadGame);

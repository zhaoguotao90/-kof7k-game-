    <script>
        // 全局变量
        let rufflePlayer = null;
        let countdown = 10;
        let countdownTimer = null;
        
        // 日志函数
        function addLog(message, type = 'info') {
            const log = document.getElementById('log');
            const time = new Date().toLocaleTimeString();
            const color = type === 'error' ? '#ff5555' : 
                         type === 'success' ? '#55ff55' : 
                         type === 'warning' ? '#ffff55' : '#aaaaaa';
            
            log.innerHTML += `<div style="color:${color}">[${time}] ${message}</div>`;
            log.scrollTop = log.scrollHeight;
            console.log(`[${type}] ${message}`);
        }
        
        function clearLog() {
            document.getElementById('log').innerHTML = '日志已清空';
        }
        
        // 加载游戏
        async function loadGame() {
            addLog('正在初始化Ruffle播放器...', 'info');
            
            // 移除旧的播放器
            if (rufflePlayer) {
                document.getElementById('gameContainer').innerHTML = '';
            }
            
            // 配置Ruffle
            window.RufflePlayer = window.RufflePlayer || {};
            window.RufflePlayer.config = {
                "autoplay": "on",
                "unmuteOverlay": "hidden",
                "backgroundColor": "#000000"
            };
            
            try {
                const ruffle = window.RufflePlayer.newest();
                rufflePlayer = ruffle.createPlayer();
                document.getElementById('gameContainer').appendChild(rufflePlayer);
                
                await rufflePlayer.load({
                    url: 'kof7k_modified.swf',  // 确保文件名与您保存的一致
                    width: 800,
                    height: 600
                });
                
                addLog('游戏加载成功！请开始游戏。', 'success');
                
            } catch (error) {
                addLog(`游戏加载失败: ${error.message}`, 'error');
            }
        }
        
        // 监听来自Flash的消息
        window.addEventListener('message', function(event) {
            if (event.data && event.data.type) {
                console.log('收到消息:', event.data);
                addLog(`收到信号: ${event.data.type}`, 'info');
                
                if (event.data.type === 'KOF_GAME_VICTORY' || 
                    event.data.type === 'KOF_WIN' ||
                    event.data.type.includes('VICTORY')) {
                    
                    addLog('🎉 检测到通关信号！', 'success');
                    
                    // 更新显示信息
                    if (event.data.winner) {
                        document.getElementById('winnerDisplay').textContent = 
                            event.data.winner === 'PLAYER_VICTORY' ? '玩家' : '电脑';
                    }
                    if (event.data.playerWins) {
                        document.getElementById('winsDisplay').textContent = event.data.playerWins;
                    }
                    
                    // 显示通关弹窗
                    showVictoryModal();
                }
            }
        });
        
        // 为ExternalInterface准备的函数
        window.onKofVictory = function(winner) {
            addLog(`通过ExternalInterface收到胜利信号: ${winner}`, 'success');
            document.getElementById('winnerDisplay').textContent = 
                winner === 'PLAYER_VICTORY' ? '玩家' : '电脑';
            showVictoryModal();
        };
        
        // 弹窗控制
        function showVictoryModal() {
            const modal = document.getElementById('victoryModal');
            modal.style.display = 'flex';
            
            // 开始倒计时
            startCountdown();
            
            addLog('通关选择弹窗已显示', 'info');
        }
        
        function hideVictoryModal() {
            document.getElementById('victoryModal').style.display = 'none';
            if (countdownTimer) {
                clearInterval(countdownTimer);
                countdownTimer = null;
            }
        }
        
        function startCountdown() {
            countdown = 10;
            const countdownElement = document.querySelector('#victoryModal p:last-child');
            
            countdownTimer = setInterval(() => {
                countdown--;
                if (countdownElement) {
                    countdownElement.textContent = `${countdown}秒后自动继续游戏...`;
                }
                
                if (countdown <= 0) {
                    clearInterval(countdownTimer);
                    continueGame();
                }
            }, 1000);
        }
        
        // 用户选择
        function showVideo() {
            addLog('用户选择观看视频', 'info');
            hideVictoryModal();
            
            const videoModal = document.getElementById('videoModal');
            const video = document.getElementById('hiddenVideo');
            
            videoModal.style.display = 'flex';
            
            // 尝试播放视频
            video.play().catch(error => {
                addLog(`视频播放失败: ${error.message}`, 'warning');
            });
            
            // 视频结束时自动关闭
            video.onended = function() {
                setTimeout(() => {
                    closeVideo();
                    addLog('视频播放完成', 'info');
                }, 1000);
            };
        }
        
        function closeVideo() {
            const videoModal = document.getElementById('videoModal');
            const video = document.getElementById('hiddenVideo');
            
            video.pause();
            video.currentTime = 0;
            videoModal.style.display = 'none';
        }
        
        function continueGame() {
            addLog('用户选择继续游戏', 'info');
            hideVictoryModal();
            
            // 重新加载游戏
            if (rufflePlayer) {
                rufflePlayer.load({
                    url: 'kof7k_modified.swf',
                    width: 800,
                    height: 600
                });
                addLog('游戏已重新加载', 'info');
            }
        }
        
        // 测试函数
        function testSignal() {
            addLog('发送测试通关信号...', 'warning');
            
            // 模拟发送通关信号
            window.postMessage({
                type: 'KOF_GAME_VICTORY',
                winner: 'PLAYER_VICTORY',
                playerWins: 2,
                roundMax: 3
            }, '*');
            
            addLog('测试信号已发送', 'success');
        }
        
        // 检查URL参数
        function checkUrlParams() {
            const params = new URLSearchParams(window.location.search);
            if (params.has('kof_victory')) {
                const winner = params.get('kof_victory');
                addLog(`从URL参数检测到通关: ${winner}`, 'warning');
                
                window.postMessage({
                    type: 'URL_VICTORY',
                    winner: winner === 'player' ? 'PLAYER_VICTORY' : 'CPU_VICTORY'
                }, '*');
            }
        }
        
        // 页面加载完成
        window.addEventListener('DOMContentLoaded', () => {
            addLog('页面加载完成', 'info');
            checkUrlParams();  // 检查URL参数
        });
    </script>
